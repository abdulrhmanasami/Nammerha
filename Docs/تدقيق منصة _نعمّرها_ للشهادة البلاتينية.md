import { NextResponse } from 'next/server';  
import crypto from 'crypto';  
import Redis from 'ioredis';  
import { PrismaClient } from '@prisma/client';

const prisma \= new PrismaClient();

// Connect exclusively to the isolated Redis instance on 91.98.182.243  
// Configured to prevent memory leaks and handle retries gracefully  
const redis \= new Redis(process.env.REDIS\_URL as string, {  
  maxRetriesPerRequest: 3,  
  enableReadyCheck: true,  
});

const FATORA\_SECRET \= process.env.FATORA\_WEBHOOK\_SECRET as string;

export async function POST(req: Request) {  
  try {  
    const rawBody \= await req.text();  
    const signature \= req.headers.get('x-fatora-signature');

    // 1\. ISO 25010 Security: Validate HMAC Signature to prevent forged financial requests  
    const expectedSignature \= crypto  
     .createHmac('sha256', FATORA\_SECRET)  
     .update(rawBody)  
     .digest('hex');

    if (signature\!== expectedSignature) {  
      console.error(' Invalid Fatora Webhook Signature Detected');  
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });  
    }

    const payload \= JSON.parse(rawBody);  
    const { transaction\_id, status, amount } \= payload;

    // Only process successful payments  
    if (\!transaction\_id |

| status\!== 'SUCCESS') {  
      return NextResponse.json({ received: true, status: 'Ignored' }, { status: 200 });  
    }

    // 2\. ISO 25010 Reliability: Distributed Locking to prevent Race Conditions & Double Spending  
    const lockKey \= \`lock:webhook:fatora:${transaction\_id}\`;  
    // 'NX' ensures the key is set only if it does not already exist. 'EX' 30 sets an auto-expire.  
    const acquiredLock \= await redis.set(lockKey, 'locked', 'EX', 30, 'NX');

    if (\!acquiredLock) {  
      // Lock exists, meaning this specific webhook transaction is already being processed by another thread.  
      console.warn(\` Duplicate webhook blocked for TX: ${transaction\_id}\`);  
      return NextResponse.json({ message: 'Already processing' }, { status: 429 });  
    }

    try {  
      // 3\. Database Transaction with Serializable Isolation Level  
      await prisma.$transaction(async (tx) \=\> {  
        const escrow \= await tx.escrow.findUnique({  
          where: { fatoraTxId: transaction\_id },  
        });

        if (\!escrow) {  
          throw new Error('Escrow account not found for specified transaction');  
        }

        if (escrow.status \=== 'RELEASED') {  
          return; // Idempotency check passed; already released previously.  
        }

        // Release funds securely ensuring fractional limits are respected  
        await tx.escrow.update({  
          where: { fatoraTxId: transaction\_id },  
          data: {  
            status: 'RELEASED',  
            releasedAmount: amount,  
          },  
        });

        // Add automated audit log for the Donor/Victim transparency dashboard   
        // This maps to the DREAM system philosophy of Radical Transparency  
        await tx.review.create({  
          data: {  
            revieweeId: escrow.beneficiaryId,  
            score: 5, // Auto-score for successful automated milestone delivery  
            metadata: {   
                event: "FUNDS\_RELEASED",   
                txId: transaction\_id,   
                proof: "FATORA\_GATEWAY",  
                timestamp: new Date().toISOString()  
            }  
          }  
        });  
      }, {  
        isolationLevel: 'Serializable', // Highest isolation level against concurrent writes/reads  
        maxWait: 5000,  
        timeout: 10000,  
      });

      return NextResponse.json({ message: 'Escrow successfully released' }, { status: 200 });  
    } finally {  
      // Always release the Redis lock to prevent deadlocks  
      await redis.del(lockKey);  
    }  
  } catch (error: any) {  
    console.error(' Fatora Webhook processing crashed:', error.message);  
    // 0% Crash Protocol: Return standard 500 without leaking stack traces or schema info   
    return NextResponse.json({ error: 'Internal Server Error' }, { status: 500 });  
  }  
}

                  double csbpSafeScore \= (10.0 \- doc.value) / 10.0;

                \`,  
                params: {  
                  lat: req.projectLat,  
                  lon: req.projectLon,  
                  maxDist: req.maxDistanceKm  
                }  
              },  
              order: 'desc' // Highest score first matches the best contractor  
            }  
          }  
        \]  
      }  
    });

    return body.hits.hits.map((hit: any) \=\> ({  
      contractorId: hit.\_source.id,  
      matchScore: hit.sort,  
      name: hit.\_source.name,  
      rating: hit.\_source.rating  
    }));  
      
  } catch (error) {  
    console.error('', error);  
    // ISO 25010 Reliability constraint: Fail securely without exposing architecture  
    throw new Error('Matchmaking algorithm failed to compile or fetch records safely.');  
  }  
}

function calculateDistance(lat1: number, lon1: number, lat2: number, lon2: number): number {  
  const R \= 6371e3; // Earth radius in meters  
  const rad \= Math.PI / 180;  
  const dLat \= (lat2 \- lat1) \* rad;  
  const dLon \= (lon2 \- lon1) \* rad;  
  const a \= Math.sin(dLat / 2) \* Math.sin(dLat / 2) \+  
            Math.cos(lat1 \* rad) \* Math.cos(lat2 \* rad) \*  
            Math.sin(dLon / 2) \* Math.sin(dLon / 2);  
  const c \= 2 \* Math.atan2(Math.sqrt(a), Math.sqrt(1 \- a));  
  return R \* c; // returns distance in meters  
}
